#include "llama.h"   // make sure this is included

// ------------------------------------------------------------------
// Structure for your engine
// ------------------------------------------------------------------
typedef struct {
    struct llama_model*   model;
    struct llama_context* ctx;
    // add other fields you need (vocab, etc.)
} engine_t;

// ------------------------------------------------------------------
// Open model using official llama.cpp loader
// ------------------------------------------------------------------
engine_t* engine_open(const char* model_path)
{
    fprintf(stderr, "[engine] loading model: %s\n", model_path);

    // Model parameters
    llama_model_params model_params = llama_model_default_params();
    // model_params.n_gpu_layers = 0;     // set >0 if you have GPU support
    // model_params.use_mmap = true;
    // model_params.use_mlock = false;

    struct llama_model* model = llama_load_model_from_file(model_path, model_params);
    if (!model) {
        fprintf(stderr, "[engine] llama_load_model_from_file failed\n");
        return NULL;
    }

    // Context parameters
    llama_context_params ctx_params = llama_context_default_params();
    ctx_params.n_ctx     = 8192;        // adjust as needed
    ctx_params.n_batch   = 512;
    ctx_params.n_threads = 0;           // 0 = auto (number of logical cores)

    struct llama_context* ctx = llama_new_context_with_model(model, ctx_params);
    if (!ctx) {
        fprintf(stderr, "[engine] llama_new_context_with_model failed\n");
        llama_free_model(model);
        return NULL;
    }

    engine_t* engine = (engine_t*)calloc(1, sizeof(engine_t));
    if (!engine) {
        llama_free(ctx);
        llama_free_model(model);
        return NULL;
    }

    engine->model = model;
    engine->ctx   = ctx;

    fprintf(stderr, "[engine] model loaded successfully!\n");
    return engine;
}

2. Close functionc

void engine_close(engine_t* engine)
{
    if (!engine) return;

    if (engine->ctx)   llama_free(engine->ctx);
    if (engine->model) llama_free_model(engine->model);

    free(engine);
}
